#include <stdio.h>

int main() {
    int matriz[3][4];
    int soma = 0;

    printf("Digite os elementos da matriz 3x4:\n");
    for (int i = 0; i < 3; i++) {
        for (int j = 0; j < 4; j++) {
            printf("Elemento [%d][%d]: ", i, j);
            scanf("%d", &matriz[i][j]);
            soma += matriz[i][j];
        }
    }

    printf("Soma de todos os elementos: %d\n", soma);
    return 0;
}

#include <stdio.h>

int main() {
    int A[3][3], B[3][3], C[3][3];

    printf("Digite os elementos da matriz A 3x3:\n");
    for (int i = 0; i < 3; i++)
        for (int j = 0; j < 3; j++) {
            printf("A[%d][%d]: ", i, j);
            scanf("%d", &A[i][j]);
        }

    printf("Digite os elementos da matriz B 3x3:\n");
    for (int i = 0; i < 3; i++)
        for (int j = 0; j < 3; j++) {
            printf("B[%d][%d]: ", i, j);
            scanf("%d", &B[i][j]);
            C[i][j] = A[i][j] + B[i][j];
        }

    printf("Matriz C (A + B):\n");
    for (int i = 0; i < 3; i++) {
        for (int j = 0; j < 3; j++)
            printf("%d ", C[i][j]);
        printf("\n");
    }

    return 0;
}

    int matriz[N][N];

    printf("Digite os elementos da matriz %dx%d:\n", N, N);
    for (int i = 0; i < N; i++)
        for (int j = 0; j < N; j++) {
            printf("Elemento [%d][%d]: ", i, j);
            scanf("%d", &matriz[i][j]);
        }

    printf("Diagonal secundária:\n");
    for (int i = 0; i < N; i++)
        printf("%d ", matriz[i][N - 1 - i]);

    printf("\n");

}
