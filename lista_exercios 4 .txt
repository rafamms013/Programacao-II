Exercício 1: Faça um programa que preencha um vetor de inteiros de tamanho 10 pedindo valores ao usuário. Em seguida, calcule e salve num segundo vetor o quadrado de cada elemento do primeiro vetor. Por fim, imprima os dois vetores.
#include <stdio.h>

int main() {
    int vetor[10];           // Primeiro vetor (entrada do usuário)
    int quadrado[10];        // Segundo vetor (quadrado dos valores)

    // Leitura dos valores do usuário
    printf("Digite 10 números inteiros:\n");
    for(int i = 0; i < 10; i++) {
        printf("Valor %d: ", i + 1);
        scanf("%d", &vetor[i]);  // Lê e armazena no vetor
        quadrado[i] = vetor[i] * vetor[i];  // Calcula e armazena o quadrado
    }

    // Impressão dos vetores
    printf("\nVetor original:\n");
    for(int i = 0; i < 10; i++) {
        printf("%d ", vetor[i]);
    }

    printf("\n\nVetor com quadrados:\n");
    for(int i = 0; i < 10; i++) {
        printf("%d ", quadrado[i]);
    }

    return 0;
}

Exercício 2: Faça um programa que some o conteúdo de dois vetores de tamanho 25 e armazene o resultado em um terceiro vetor. Imprima os três vetores na tela.
#include <stdio.h>

int main() {
    int vetor1[25], vetor2[25], soma[25];

    printf("Digite 25 valores para o primeiro vetor:\n");
    for(int i = 0; i < 25; i++) {
        printf("vetor1[%d]: ", i);
        scanf("%d", &vetor1[i]);
    }

    printf("\nDigite 25 valores para o segundo vetor:\n");
    for(int i = 0; i < 25; i++) {
        printf("vetor2[%d]: ", i);
        scanf("%d", &vetor2[i]);
        soma[i] = vetor1[i] + vetor2[i];  // Soma os valores diretamente
    }

    // Impressão dos vetores
    printf("\nVetor 1:\n");
    for(int i = 0; i < 25; i++) {
        printf("%d ", vetor1[i]);
    }

    printf("\n\nVetor 2:\n");
    for(int i = 0; i < 25; i++) {
        printf("%d ", vetor2[i]);
    }

    printf("\n\nVetor Soma:\n");
    for(int i = 0; i < 25; i++) {
        printf("%d ", soma[i]);
    }

    return 0;
}

3: Faça um programa que gere 10 números inteiros e os armazene em um vetor. Em seguida, leia um código inteiro e faça uma das ações abaixo: 0 - finaliza o programa. 1 - imprime o vetor na ordem do início ao fim. 2 - imprime o vetor na ordem inversa (do fim para o início).
#include <stdio.h>

int main() {
    int vetor[10], codigo;

    printf("Digite 10 números inteiros:\n");
    for(int i = 0; i < 10; i++) {
        printf("Valor %d: ", i + 1);
        scanf("%d", &vetor[i]);
    }

    // Leitura do código
    printf("\nDigite o código (0 = sair, 1 = ordem normal, 2 = ordem inversa): ");
    scanf("%d", &codigo);

    switch(codigo) {
        case 0:
            printf("Programa finalizado.\n");
            break;
        case 1:
            printf("Vetor na ordem normal:\n");
            for(int i = 0; i < 10; i++) {
                printf("%d ", vetor[i]);
            }
            break;
        case 2:
            printf("Vetor na ordem inversa:\n");
            for(int i = 9; i >= 0; i--) {
                printf("%d ", vetor[i]);
            }
            break;
        default:
            printf("Código inválido.\n");
    }

    return 0;
}
